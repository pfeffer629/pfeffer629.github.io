Mark & Jay's JavaScript Cheatsheet

Useful Methods

Objects

var foo = { one: 1, two: 2};

foo.hasOwnProperty('one'); // true

foo.isPrototypeOf("aString"); // false

Arrays

['red', 'blue', 'green'].length; // 3

var foo = ['red', 'blue'];

foo.push('green'); // 3

foo; // ['red', 'blue', 'green']

foo.pop(); // 'green'

foo; // ['red', 'blue']

foo.shift>(); // 'red'

foo; // ['blue']

['red', 'blue', 'green'].reverse(); // [ 'green', 'blue', 'red' ]

[0, 1, 5, 10, 15].sort() // [0, 1, 10, 15, 5] sorts in STRING order

[0, 1, 5, 10, 15].sort(function (a, b) { (a > b) ? a : b; }); // [0, 1, 5, 10, 15]

[1, 2, 3].concat([4, 5, 6]) // [1, 2, 3, 4, 5, 6]

var foo = ['red', 'blue', 'green'];

foo.splice(0, 1); // 'red' starting at 0, removes 1 elem

foo; // ['blue', 'green']

var bar = foo.splice(1, 0, 'yellow', 'orange'); // starting at 1, inserts elems, removes 0

bar; // ['blue', 'yellow', 'orange', 'green']

bar.splice(1, 1, 'red', 'purple') // 'yellow'// starting at 1, inserts elems, removes 1

bar; // ['blue', 'red', 'purple', 'orange', 'green']

[1, 2, 0, 4].every(function (x) { return x > 0; }); //false

[0, 1, 0, 0].some(function (x) { return x > 0; }); //true

var foo = [1, 2, 3, 4];

foo.filter(function (x) { return x % 2 === 0; }) // [2, 4]

foo.forEach(function (x) { console.log(x); }) // prints 1 through 4

foo.map(function (x) { return x * 2; }); // [2, 4, 6, 8]

foo.reduce(function sum(acc, next) { return acc + next }); // 10

Strings

var foo = ' hello ' + 'world '; //' hello world '

foo = foo.trim() // 'hello world'

foo[1]; // 'e'

foo.charCodeAt(1); // 101 (ASCII value for letter at index 1)

foo.length; // 11

foo.toUpperCase(); // "HELLO WORLD"

"HELLO WORLD".toLowerCase(); //"hello world"

foo.slice(3); // 'lo world'

foo.slice(3, 7); // 'lo w'

foo.slice(-3); // 'rld'

foo.indexOf('o') // 4

foo.lastIndexOf('o') // 7

Math

Math.max(3, 54, 32, 15); // 54

Math.min(3, 54, 32, 15); // 3

Math.ceil(2.7); // 3

Math.floor(2.7); // 2

Math.round(2.5); // 3

Math.round(2.4); // 2

Math.abs(-5); // 5

Math.pow(2, 3); // 8

Math.sqrt(16); // 4

Gotchas

  • Use var with all variable declarations. Otherwise the variable will be global, no matter where it's declared.

  • Beyond functions, JS has no block scope. A variable declared in a loop is available everywhere in the containing function (or globally if there isn't one).

  • Failing to use new when calling an object constructor will cause all instances of this within the object to refer to the global object.

  • Use === and !== to test for equality. (== and != use type conversions that follow esoteric rules.

  • Use the following to test whether value is an array:

    Array.isArray(value)

  • Use the following to test whether value is a number:

    function isNumber(value) { return typeof value === 'number' && isFinite(value);};

  • Use the following to test whether value is an object:

    value && typeof value === 'object'

  • for in statements loop through not only the properties on an object, but any properties inherited through the prototype chain. Use the following to filter out prototype properties:

    for (name in object) { if (object.hasOwnProperty(name)) { /* do something */ }}

  • Be careful with floating point arithmetic: 0.1 + 0.2 = 0.30000000000000004

  • Be careful with +. If operands are both numbers, it performs addition. If both are strings, it concatenates them. But if the types are mixed, it concatenates them after converting them both to strings.

  • A return statement must have its argument begin on the same line, otherwise JS will automatically insert a semicolon, causing it to return undefined
